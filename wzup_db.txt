CREATE TABLE `Users` (
  `UserId` INT NOT NUll auto_increment,
  `CompanyName` VARCHAR(30),
  `Phone` INT,
  `EMail` VARCHAR(30),
  PRIMARY KEY (`UserId`)
);


CREATE TABLE `Days` (
  `DayID` INT NOT NULL,
  `Weekday` VARCHAR(10),
  PRIMARY KEY (`DayID`)
)ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_danish_ci;

CREATE TABLE `Events` (
  `EventId` INT NOT NULL auto_increment,
  `UserId` INT,
  `Header` VARCHAR(30),
  `Text` VARCHAR(50),
  `Location` VARCHAR(30),
  PRIMARY KEY (`EventId`),
  FOREIGN KEY (`UserId`) REFERENCES users (`UserId`)
);

CREATE TABLE `EventTime` (
  `EventId` INT,
  `Date` DATE,
  `DayId` INT,
  `StartTime` TIME,
  `EndTime` TIME,
  PRIMARY KEY (`Date`, `EventId`),
  FOREIGN KEY (`EventId`) REFERENCES events (`EventId`),
  FOREIGN KEY (`DayId`) REFERENCES days (`DayID`)
);
